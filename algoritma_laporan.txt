Bisa, bos. Ini algoritma yang menurutku paling konsisten dengan data Garmin FIT yang tidak memiliki cadence. Engine ini sudah disesuaikan dengan tinggi badanmu (166.5 cm) dan gaya lari short stride.

# ==========================================
# CADENCE & STRIDE ESTIMATION ENGINE v2
# Athlete Height : 166.5 cm
# ==========================================

ATHLETE_HEIGHT = 166.5

def estimate_cadence(pace_min_km):
    """
    pace_min_km : menit/km
    return : cadence (SPM), mode
    """

    if pace_min_km < 5.12:
        cadence = 174
        mode = "RUN (Z6)"

    elif pace_min_km < 5.45:
        cadence = 171
        mode = "RUN (Z5)"

    elif pace_min_km < 5.82:
        cadence = 169
        mode = "RUN (Z4)"

    elif pace_min_km < 6.48:
        cadence = 166
        mode = "RUN (Z3)"

    elif pace_min_km < 7.55:
        cadence = 160
        mode = "RUN (Z2)"

    else:
        cadence = 115
        mode = "WALK (Z1)"

    # koreksi tinggi badan
    cadence *= (170 / ATHLETE_HEIGHT)

    return round(cadence), mode

def estimate_stride(speed_ms, cadence):

    stride = speed_ms / (cadence / 60)

    # koreksi agar mendekati data runner sungguhan
    if cadence >= 160:
        stride *= 1.04

    return round(stride * 100)      # cm
